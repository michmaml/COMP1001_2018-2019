CREATE TABLE User_addresses (
UserID INTEGER NOT NULL REFERENCES User_details(UserID),
Keyword TEXT NOT NULL,
Address TEXT NOT NULL,
PRIMARY KEY(UserID, Keyword)
);
CREATE TABLE Orders (
OrderID INTEGER NOT NULL,
CarID INTEGER NOT NULL REFERENCES Cars(CarID),
UserID INTEGER NOT NULL REFERENCES User_details(UserID),
Twitter_handle TEXT NOT NULL,
Pickup_location TEXT NOT NULL,
Date TEXT NOT NULL,
Time TEXT NOT NULL,
Status INTEGER NOT NULL,
PRIMARY KEY(OrderID)
);
CREATE TABLE Tweets (
TweetID INTEGER NOT NULL,
OrderID INTEGER NOT NULL REFERENCES Orders(OrderID),
Status INTEGER NOT NULL,
Reply TEXT,
PRIMARY KEY(TweetID)
);
CREATE TABLE User_details (
UserID INTEGER NOT NULL,
Firstname TEXT NOT NULL,
Surname TEXT NOT NULL,
Twitter_handle TEXT NOT NULL UNIQUE,
Password TEXT NOT NULL,
Email TEXT NOT NULL UNIQUE,
Phone_number INTEGER,
Reward_points INTEGER,
Total_orders INTEGER,
Total_cancellations INTEGER,
Status INTEGER NOT NULL,
PRIMARY KEY(UserID)
);
CREATE TABLE Car_status (
Status INTEGER NOT NULL REFERENCES Cars(Status),
Description TEXT NOT NULL UNIQUE,
PRIMARY KEY(Status)
);
CREATE TABLE Car_types (
Type INTEGER NOT NULL REFERENCES Cars(Type),
Description TEXT NOT NULL UNIQUE,
PRIMARY KEY(Type)
);
CREATE TABLE Order_status (
Status INTEGER NOT NULL REFERENCES Orders(Status),
Description TEXT NOT NULL UNIQUE,
PRIMARY KEY(Status)
);
CREATE TABLE Cars(
CarID INTEGER NOT NULL,
Status INTEGER NOT NULL,
Type INTEGER NOT NULL,
Seats INTEGER NOT NULL,
Location TEXT NOT NULL,
PRIMARY KEY (CarID));
