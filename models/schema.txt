CREATE TABLE Cars (
CarID INTEGER,
Status INTEGER,
Type INTEGER,
Seats INTEGER,
PRIMARY KEY(CarID)
);
CREATE TABLE Orders (
OrderID INTEGER,
CarID INTEGER,
UserID INTEGER,
Pickup_location TEXT,
Date INTEGER,
Time INTEGER,
PRIMARY KEY(OrderID, CarID, UserID)
);
CREATE TABLE Userdetails (
UserID INTEGER NOT NULL,
Firstname TEXT NOT NULL,
Surname TEXT NOT NULL,
Twitter_handle TEXT NOT NULL UNIQUE,
Username TEXT NOT NULL UNIQUE,
Password TEXT NOT NULL,
Reward_points INTEGER,
Account_deactivated INTEGER,
Total_orders INTEGER,
Total_cancellations INTEGER,
PRIMARY KEY(UserID)
);
CREATE TABLE User_addresses (
UserID INTEGER NOT NULL REFERENCES Userdetails(UserID),
Address TEXT NOT NULL,
Keyword TEXT NOT NULL,
PRIMARY KEY(UserID, Address)
);
CREATE TABLE User_phone_number (
UserID INTEGER NOT NULL REFERENCES Userdetails(UserID),
Phone_number INTEGER NOT NULL,
PRIMARY KEY(UserID, Phone_number)
);
